        function convertMarkdownToHtml(markdown) {
            let html = markdown;
            
            // í—¤ë” ë³€í™˜
            html = html.replace(/^### (.*$)/gm, '<h3>$1</h3>');
            html = html.replace(/^## (.*$)/gm, '<h2>$1</h2>');
            html = html.replace(/^# (.*$)/gm, '<h1>$1</h1>');
            
            // êµµì€ ê¸€ì”¨ ë³€í™˜
            html = html.replace(/\*\*(.*?)\*\*/g, '<strong>$1</strong>');
            
            // ë¦¬ìŠ¤íŠ¸ ë³€í™˜

            html = html.replace(/(<li>.*<\/li>)/s, '<ul>$1</ul>');
            
            // ì¤„ë°”ê¿ˆ ë³€í™˜
